create database TresDeMaio_DB;
use TresDeMaio_DB;

create table Socio(
   Id INT NOT NULL AUTO_INCREMENT,
   Ativo bool,
   Titulo Int not null,
   CPF VARCHAR(14) unique,
   Nome VARCHAR(200) NOT NULL,
   Rg varchar(15),
   OrgaoExpedidor varchar(50),
   UFOrgaoExpedidor char(2),
   DataExpedicao datetime,
   DataNascimento datetime,
   DataAdesao datetime,
   DataCadastro datetime,
   DataAtualizacao datetime,
   FoneResidencial varchar(16),
   FoneCelular varchar(16),
   FoneComercial varchar(16),
   Email varchar(200),
   Obs varchar(1000),
   Situacao varchar(200),
   UltPagamento varchar(20),   
   PathImagem varchar(500),
   PRIMARY KEY ( Id )
);

create table Endereco(
	Id INT NOT NULL AUTO_INCREMENT,
    Cep VARCHAR(12),
    Rua VARCHAR(200) NOT NULL,
    Numero varchar(8),
    Bairro varchar(100),
    Cidade varchar(150),
    UF char(2),
    Complemento varchar(200),
    IdSocio Int not null,
    PRIMARY KEY ( Id ),
    FOREIGN KEY (IdSocio) REFERENCES Socio(Id)
);

create table Dependente(
	Id INT NOT NULL AUTO_INCREMENT,
    Cpf VARCHAR(14),
    Nome VARCHAR(200) NOT NULL,
    DataNascimento datetime,
    Parentesco varchar(100),
    Numero int,
    Obs varchar(500),   
    Fone varchar(16),
    DataInclusao datetime,
    IdSocio Int not null,
    PRIMARY KEY ( Id ),
    FOREIGN KEY (IdSocio) REFERENCES Socio(Id)
);

create table Usuarios(
	Id Int not null auto_increment,
    Matricula varchar(100),
    Usuario varchar(200) not null,
    Senha varchar(50),
    PSocios bool,
    PReceitas bool,
    PRelatorios bool,
    FIncluirSocios bool,
    FCadastroReceitas bool,
    FBaixaReceitas bool,
    IsAdministrator bool,
    PRIMARY KEY (Id)    
);

create table Receitas(
	Id INT NOT NULL AUTO_INCREMENT,
    Documento int not null,
    Parcela int,
    DataVencimento datetime not null,
    DiaVencimento int,
    Valor double,
    FlagPago boolean,
    Obs varchar(500),  
    DataCadastro datetime,
    IdSocio Int not null,
    PRIMARY KEY ( Id ),
    FOREIGN KEY (IdSocio) REFERENCES Socio(Id)
);

Create table Pagamentos(
	Id int not null auto_increment,
    Documento long not null,
    DataVencimento datetime,
    Valor double not null,
    Desconto double,
    Juros double,
    ValorDescJuros double,
    ValorPago double,
    DataPagamento datetime,
    ObsPagamento varchar(500),
    IdSocio Int not null,    
    IdReceita int not null,
    PRIMARY KEY ( Id ),
    FOREIGN KEY (IdSocio) REFERENCES Socio(Id),
    FOREIGN KEY (IdReceita) REFERENCES Receitas(Id)
);

create table Sede(
	Id Int not null auto_increment,
    NomeFantasia varchar(300),
    RazaoSocial varchar(300),
    Cnpj varchar(20),
    InscricaoEstadual varchar(50),
    Endereco varchar(200),
    Numero varchar(20),
    Bairro varchar(200),
    Cidade varchar(200),
    UF varchar(2),
    Telefone varchar(20),
    Email varchar(200),
    PRIMARY KEY (Id)    
);
